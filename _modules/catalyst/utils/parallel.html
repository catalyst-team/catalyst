[docs]def tqdm_parallel_imap(
    func,
    args,
    pool: Union[Pool, DumbPool],
    total: int = None,
    pbar=tqdm,
) -> List[T]:
    if total is None and hasattr(args, "__len__"):
        total = len(args)

    if pbar is None:
        result = parallel_imap(func, args, pool)
    else:
        result = list(pbar(pool.imap_unordered(func, args), total=total))

    return result
